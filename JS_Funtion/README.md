## ✔ 선언과 표현, 호이스팅

```javascript
// 함수 선언문(Declaration)
function hello() {}

// 함수 표현식(Expression)
const hello = function () {};

// 호출방법 : 함수이름을 적고 소괄호로 호출하여 실행한다.
hello();

------------------------------------------------------------

// 호이스팅(Hoisting)

```
